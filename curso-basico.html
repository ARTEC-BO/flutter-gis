        .map((p) => p['altitud'] as double? ?? 0)

      'total': puntos.length,
      'sincronizados': sincronizados,
      'pendientes': puntos.length - sincronizados,
      'altitud_promedio': altitudPromedio.toStringAsFixed(1),
      'altitud_minima': altitudMin.toStringAsFixed(1),
      'altitud_maxima': altitudMax.toStringAsFixed(1),
